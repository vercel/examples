  3: 10,
  4: 9,
  5: 7.5,
}

for await (const country of COUNTRIES) {
  let factor = 1

  try {
    factor = await fetch(
      `https://api.purchasing-power-parity.com/?target=${country.toUpperCase()}`
    )
      .then((res) => res.json())
      .then(({ ppp: { pppConversionFactor: factor } }) => factor)
  } catch (e) {
    console.warn(`Failed for ${country}`)
  }

  prices[country] = Object.entries(REGIONS).reduce(
    (match, item) => (PRICE * factor < match[1] ? item : match),
    ['1', Infinity]
  )[0]
}

console.log(prices)
